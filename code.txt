            function listenForData() {
                onSnapshot(jobsCollection, (snapshot) => {
                    jobs = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
                    if(!isInitialDataLoaded.jobs) { isInitialDataLoaded.jobs = true; checkInitialLoad(); }
                    else { renderCurrentView(); }
                }, error => console.error("Jobs listener error: ", error));

                onSnapshot(engineersCollection, (snapshot) => {
                    engineers = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
                    if(!isInitialDataLoaded.engineers) { isInitialDataLoaded.engineers = true; checkInitialLoad(); }
                    else { renderCurrentView(); }
                }, error => console.error("Engineers listener error: ", error));

                onSnapshot(settingsDocRef, (doc) => {
                    if (doc.exists()) {
                        const data = doc.data();
                        categories = data.categories || [];
                        specialisations = data.specialisations || [];
                    } else {
                        setDoc(settingsDocRef, { 
                            categories: ['Installation', 'Survey', 'Remedial', 'Maintenance', 'Holiday', 'Other'],
                            specialisations: [...specialisationOrder, 'Admin']
                        }).catch(e => console.error("Error creating default settings:", e));
                    }
                    if(!isInitialDataLoaded.settings) { isInitialDataLoaded.settings = true; checkInitialLoad(); }
                    else { renderCurrentView(); }
                }, error => console.error("Settings listener error: ", error));

            const categoryColours = { 'Installation': 0, 'Survey': 120, 'Remedial': 30, 'Maintenance': 200, 'Holiday': 280, 'Other': 50 };
            const getColourByCategory = (category, jobId) => {
                const baseHue = categoryColours[category] || 0;
                let hash = 0;
                for (let i = 0; i < jobId.length; i++) { hash = jobId.charCodeAt(i) + ((hash << 5) - hash); }
                const lightness = 85 + (hash % 10);
                const saturation = 65 + (hash % 10);
                return { bgColour: `hsl(${baseHue}, ${saturation}%, ${lightness}%)`, borderColour: `hsl(${baseHue}, ${saturation-10}%, ${lightness-10}%)` };
            };

            function renderSchedule() {
                // FIX: Implemented correct Monday-start logic for week view
                const { start, end } = currentView === 'week' 
                    ? (() => { 
                        const s = new Date(currentDate);
                        s.setHours(12,0,0,0); // Avoid timezone shifts around midnight
                        const day = s.getDay();
                        const diff = s.getDate() - day + (day === 0 ? -6 : 1); // adjust when day is sunday
                        s.setDate(diff);
                        const e = new Date(s);
                        e.setDate(s.getDate() + 6);
                        return { start: s, end: e }; 
                      })()
                    : { start: new Date(currentDate.getFullYear(), currentDate.getMonth(), 1), end: new Date(currentDate.getFullYear(), currentDate.getMonth() + 1, 0) };

                dom.currentDateRange.textContent = currentView === 'week'
                    ? `${start.toLocaleDateString('en-GB', { day: 'numeric', month: 'short' })} - ${end.toLocaleDateString('en-GB', { day: 'numeric', month: 'short', year: 'numeric' })}`
                    : `${start.toLocaleDateString('en-GB', { month: 'long', year: 'numeric' })}`;

                const daysInView = [];
                let d = new Date(start);
const endCopy = new Date(end);
while (d <= endCopy) {
    daysInView.push(new Date(d)); // push a true copy!
    d.setDate(d.getDate() + 1);
}

                
                dom.scheduleGrid.innerHTML = '';
                dom.scheduleGrid.style.gridTemplateColumns = `12rem repeat(${daysInView.length}, minmax(8rem, 1fr))`;
                dom.scheduleGrid.style.display = 'grid';

                dom.scheduleGrid.insertAdjacentHTML('beforeend', `<div class="p-4 font-bold text-left sticky top-0 left-0 bg-gray-100 z-20 border-b border-r">Engineer</div>`);
                const todayString = formatDate(new Date());
                daysInView.forEach(day => {
                    const dateString = formatDate(day);
                    let classes = 'p-2 font-bold text-center sticky top-0 z-10 border-b border-r ';
                    if (dateString === todayString) classes += 'today-highlight ';
                    if (bankHolidays.has(dateString)) classes += 'bank-holiday ';
                    else if ([0, 6].includes(day.getDay())) classes += 'weekend-day ';
                    else if (!classes.includes('today-highlight')) classes += 'bg-gray-50 ';
                    dom.scheduleGrid.insertAdjacentHTML('beforeend', `<div class="${classes}" title="${bankHolidays.get(dateString) || ''}">${day.toLocaleDateString('en-GB', { weekday: 'short' })}<br><span class="font-normal text-sm">${day.getDate()}</span></div>`);
                });

                // FIX: Ensured engineer sorting is applied correctly every time.
                const sortedSpecialisations = [...new Set(engineers.map(e => e.type))].sort((a, b) => {
                    const indexA = specialisationOrder.indexOf(a);
                    const indexB = specialisationOrder.indexOf(b);
                    if (indexA === -1 && indexB === -1) return a.localeCompare(b);
                    if (indexA === -1) return 1;
                    if (indexB === -1) return -1;
                    return indexA - indexB;
                });
                
                sortedSpecialisations.forEach(type => {
                    const activeEngineersOfType = engineers.filter(e => e.type === type && e.isActive);
                    if (activeEngineersOfType.length === 0) return;

                    dom.scheduleGrid.insertAdjacentHTML('beforeend', `<div class="specialisation-header">${type}</div>`);
                    activeEngineersOfType.sort((a,b) => a.name.localeCompare(b.name)).forEach(engineer => {
                        const nameCell = document.createElement('div');
                        nameCell.className = 'p-2 font-medium sticky left-0 bg-white z-10 border-b border-r flex items-center';
                        nameCell.textContent = engineer.name;
                        dom.scheduleGrid.appendChild(nameCell);

                        daysInView.forEach(day => {
                            const dateString = formatDate(day);
                            const cell = document.createElement('div');
                            cell.className = 'border-b border-r drop-zone';
                            if (dateString === todayString) cell.classList.add('today-highlight');
                            if (bankHolidays.has(dateString)) cell.classList.add('bank-holiday');
                            else if ([0, 6].includes(day.getDay())) cell.classList.add('weekend-day');
                            
                            cell.dataset.date = dateString;
                            cell.dataset.engineerId = engineer.id;
                            
                            jobs.filter(j => j.engineerIds.includes(engineer.id)).forEach(job => {
                                const startDate = parseDate(job.startDate);
                                const endDate = parseDate(job.endDate);
                               if (dateOnly(day) >= dateOnly(startDate) && dateOnly(day) <= dateOnly(endDate)) {
                                    const dayOfWeek = day.getDay();
                                    if (job.allowWeekendWork || (dayOfWeek !== 0 && dayOfWeek !== 6)) {
                                         cell.appendChild(createJobCard(job));
                                    }
                                }
                            });
                            dom.scheduleGrid.appendChild(cell);
                        });
                    });
                });
                addDragAndDropListeners();
            }

            function addDragAndDropListeners() {
                document.querySelectorAll('.job-card').forEach(card => {

                        const job = jobs.find(j => j.id === e.target.dataset.jobId);

                    zone.addEventListener('dragover', e => { e.preventDefault(); zone.classList.add('drop-zone-hover'); });
                    zone.addEventListener('dragleave', () => zone.classList.remove('drop-zone-hover'));
                    zone.addEventListener('drop', async (e) => {
                        e.preventDefault();
                        const dropTarget = e.target.closest('.drop-zone');
                        dropTarget.classList.remove('drop-zone-hover');
                        if (draggedJobInfo && dropTarget) {
                            const { job, duration } = draggedJobInfo;
                            const newStartDate = parseDate(dropTarget.dataset.date);
                            const newEndDate = new Date(newStartDate);
                            newEndDate.setDate(newStartDate.getDate() + duration);
                            
                            const updatedData = {

                                engineerIds: [dropTarget.dataset.engineerId]
                            };
                            await updateDoc(doc(jobsCollection, job.id), updatedData);
                        }
                    });
                });
            }

            function renderForecastChart() {
                const year = new Date().getFullYear();
                dom.mainTitle.textContent = `PFS Network Ltd - Forecast View - ${year}`;
                const currentWeek = getWeekNumber(new Date());

                const table = dom.forecastTable;
                table.innerHTML = ''; // Clear previous content

                // Create Header
                const thead = table.createTHead();
                const headerRow = thead.insertRow();
                let headerHTML = `<th class="forecast-header">Project (Job No.)</th>`;
                for (let i = 1; i <= 53; i++) { 
                    headerHTML += `<th class="forecast-header ${i === currentWeek ? 'current-week-highlight' : ''}" data-week="${i}">W${i}</th>`;
                }
                headerRow.innerHTML = headerHTML;

                // Create Body
                const tbody = table.createTBody();
                const jobsInYear = jobs.filter(job => job.jobNo && (parseDate(job.startDate).getFullYear() === year || parseDate(job.endDate).getFullYear() === year));
                const projects = jobsInYear.reduce((acc, job) => {
                    const jobNo = job.jobNo.trim();
                    if (jobNo === '') return acc;
                    if (!acc[jobNo]) {
                        acc[jobNo] = { 
                            company: job.company, 
                            site: job.site,
                            totalCost: 0,
                            latestEndDate: parseDate('1970-01-01'),
                            isScheduled: true
                        };
                    }
                    acc[jobNo].totalCost += parseFloat(job.cost || 0);
                    const endDate = parseDate(job.endDate);
                    if (endDate > acc[jobNo].latestEndDate) {
                        acc[jobNo].latestEndDate = endDate;
                    }
                    if(!job.isScheduled) acc[jobNo].isScheduled = false;
                    return acc;
                }, {});

                const weeklyTotals = Array(53).fill(0);

                Object.entries(projects).sort((a,b) => a[0].localeCompare(b[0])).forEach(([jobNo, project]) => {
                    const row = tbody.insertRow();
                    const endWeek = getWeekNumber(project.latestEndDate);

                    let rowHTML = `<td title="${project.company} - ${project.site} (${jobNo})">${project.company} - ${project.site} (${jobNo})</td>`;
                    for (let i = 1; i <= 53; i++) {
                        if (i === endWeek && project.totalCost > 0) {
